Usage

  <!DOCTYPE html>
  <html>
  <head>

  <!-- add to your head -->
  <link rel=StyleSheet href="jquery-wm.css" TYPE="text/css" MEDIA=screen>
  <script type='text/javascript' src='jquery.js'></script>
  <script type='text/javascript' src='jquery-wm.min.js'></script>

  </head>

Open child windows using jquery object. 

      var $w = $.WM_open('http://retrogames.com');

      // other jquery extensions that can be used on children windows
      $w.WM_maximize();
      $w.WM_minimize();
      $w.WM_raise();
      $w.WM_close();

      // open a window and manipulate the window content
      var $w = $.WM_open();
      $w.find('.titlebartext').text('foo');
      $w.find('.windowcontent').html('here is some content');
